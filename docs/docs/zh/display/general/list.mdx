# List 列表

> 通用列表组件，支持平面列表和分组列表，点击高亮选中

## 示例效果
<div>
    <ListDemo/>
</div>
```ts title="代码 示例"
import React, { useState } from "react";
import { List } from '@matechat/react';

export default function ListDemoFlat() {
  const [selected, setSelected] = useState<string | undefined>();

  const options = [
    { label: "选项 1", value: "1" },
    { label: "选项 2", value: "2" },
    { label: "选项 3", value: "3" },
  ];

  return (
    <div
      style={{
        width: 400,
        margin: "0 auto",
        background: "#fafafa",
        border: "1px solid #eee",
        borderRadius: 8,
        padding: 16,
        boxShadow: "0 2px 6px rgba(0,0,0,0.05)"
      }}
    >
      <div style={{ marginBottom: 12, fontWeight: 500, fontSize: 14 }}>
        当前选中:{" "}
        <span style={{ color: selected ? "#2563eb" : "#999" }}>
          {selected || "无"}
        </span>
      </div>
      <List value={selected} options={options} onSelected={setSelected} />
    </div>
  );
}

```

## 参数
| Prop                  | Type                                      | Default   | Description     |
| --------------------- | ----------------------------------------- | --------- | --------------- |
| `value`               | `string \| undefined`                     | –         | 当前选中的值          |
| `options`             | `SelectOptionsType`                       | –         | 数据源，支持平面或分组     |
| `optionGroupChildren` | `string`                                  | `"items"` | 分组内选项字段名        |
| `optionGroupLabel`    | `string`                                  | `"label"` | 分组标题字段名         |
| `optionGroupTemplate` | `(group: OptionGroup) => React.ReactNode` | –         | 自定义分组标题渲染       |
| `optionLabel`         | `string`                                  | `"label"` | option 显示文本字段   |
| `optionValue`         | `string`                                  | `"value"` | option value 字段 |
| `onSelected`          | `(value: string) => void`                 | –         | 点击选项触发          |
